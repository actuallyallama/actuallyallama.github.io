Currently Unnamed Language

Statically typed scripting language.
Please note: This is a WIP. Everything is subject to change.

Example

fn Main(args:string[]) -> int {
    args.Map(x => x.Reverse()).Foreach(x => print(x))
}

fn Map <T> ( src:T[]:this, Action:T->T ) -> T[] {
    newlist:T[] = new List<T>(src.len)
    foreach val:T, index:int in src {
        newlist[index] = Action(val)
    }
}
fn Foreach <T> ( src:T[]:this, Action:T->void ) -> void {
    foreach val:T in src {
        Action(val)
    }
}
